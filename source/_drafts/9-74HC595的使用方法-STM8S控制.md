```c
#include <stdint.h>
#include "stm8s.h"

#define __nop() __asm nop __endasm

unsigned char data = 0x01;

void delay() {
  uint32_t i;
  for(i = 0; i < 5000; i++)
    __asm nop __endasm;
}

void setup()
{
  PD_DDR = 0xFF;
  PD_CR1 = 0xFF;
  PD_CR2 = 0x00;
}

void loop()
{
  unsigned char i;
  for(i=0; i<8; i++)
  {
    if((data >> i) & 0x01)
      PD_ODR |= 0x08;
    else
      PD_ODR &= 0xF7;
    PD_ODR &= 0xFB;
    __nop(); __nop(); __nop(); __nop(); __nop();
    PD_ODR |= 0x04;
  }
  PD_ODR &= 0xFD;
  __nop(); __nop(); __nop(); __nop(); __nop();
  PD_ODR |= 0x02;

  data <<= 1;
  if(data == 0) data = 0x01;
  delay();
}

void main()
{
  setup();
  while(1) loop();
}
```

```bash
// 编译
sdcc -mstm8 -lstm8 --out-fmt-ihx -o build/ main.c
// 烧写
sudo stm8flash -c stlinkv2 -p stm8s103f3 -w build/main.ihx
```
